#储存用户信息
create table if not exists `user_info`(
    `id`  int  auto_increment  primary key ,//自增id主键
    `openid`    char(40)  not null ,//微信openid
    `update_time`  bigint    not null ,//账户更新时间
    `status`       tinyint(1)   default 1,1//1-正常，0-删除
    `email`     varchar(50) ,//邮箱
    `tel`       char(11)  ,//电话
    index openid_index (openid)//普通索引
)engine=innoDB  default charset=utf8;

#储存设备信息　分表
device_info_(0~f)　
create table if not exists `device_info_f`(
    `id`  int  auto_increment  primary key ,
    `openid` char(40)  not null ,
    `name` char(20), //设备名
    `device_id`  char(20)    not null ,//设备id
    `status`       tinyint(1)   default 1,//1-正常，0-删除 2-微信报警
    `type`     tinyint(1),设备类型
    index openid_index (openid)
)engine=innoDB  default charset=utf8;
